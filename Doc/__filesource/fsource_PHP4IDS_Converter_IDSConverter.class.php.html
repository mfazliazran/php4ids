  103.       * Checks for common charcode pattern and decodes them
  104.       *
  105.       * @static

  108.       * @return  string $value
  109.       */
  110.     function ConvertFromJSCharcode($value)
  111.     {
  112.         # check if value matches typical charCode pattern
  113.         if (preg_match_all('/(?:[\w+-=\/\* ]*(?:\s?,\s?[\w+-=\/\* ]+)+)/s', $value, $matches))
  114.         {
  115.             $converted='';
  116.             $string=implode(',', $matches[0]);
  117.             $string=preg_replace('/\s/', '', $string);
  118.             $string=preg_replace('/\w+=/', '', $string);
  119.             $charcode=explode(',', $string);
  120.  
  121.             foreach($charcode as $char)
  122.             {
  123.                 $char=preg_replace('/[\W]0/s', '', $char);
  124.                 if(preg_match_all('/\d*[+-\/\* ]\d+/', $char, $matches))
  125.                 {
  126.                     $match=preg_split('/([\W]?\d+)/', (implode('', $matches[0])), null, PREG_SPLIT_DELIM_CAPTURE);
  127.                     $converted.=chr(array_sum($match));
  128.  
  129.                 }
  130.                 else if(!empty($char)) $converted.=chr($char);
  131.             }
  132.             $value.="\n[" . $converted . "] ";
  133.         }
  134.  
  135.         # check for octal charcode pattern
  136.         if (preg_match_all('/(?:(?:[\\\]+\d+\s*){2,})/ims', $value, $matches))
  137.         {
  138.             $converted='';
  139.             $charcode=explode('\\', preg_replace('/\s/', '', implode(',', $matches[0])));
  140.             foreach($charcode as $char)
  141.             {
  142.                 if(!empty($char)) $converted .= chr(octdec($char));
  143.             }
  144.             $value.="\n[" . $converted . "] ";
  145.         }
  146.  
  147.         # check for hexadecimal charcode pattern
  148.         if (preg_match_all('/(?:(?:[\\\]+\w+\s*){2,})/ims', $value, $matches))
  149.         {
  150.             $converted='';
  151.             $charcode=explode('\\', preg_replace('/[ux]/', '', implode(',', $matches[0])));
  152.             foreach($charcode as $char)
  153.             {
  154.                 if(!empty($char)) $converted .= chr(hexdec($char));
  155.             }
  156.             $value.="\n[" . $converted . "] ";
  157.         }
  158.         return $value;
  159.     }

  214.             '/("\s*[\W]+\s*\n*")*/ms',
